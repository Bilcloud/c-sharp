		static double itemCostDiscount(double purchasedNumber, double totalCost)
		{
			double result;
			if (7 < purchasedNumber && purchasedNumber < 10)
			{
				result = totalCost - 30;
			}
			else if (purchasedNumber >= 10)
			{
				result = totalCost - totalCost / 2;
			}
			else
			{
				result = totalCost;
			}
			return result;
